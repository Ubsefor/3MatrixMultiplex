26 void init_array( int ni, int nj, int nk, int nl, int nm,
27  float A[ni][nk], float B[nk][nj],
28  float C[nj][nm], float D[nm][nl] )
29 {
30  int i, j;
31  omp_set_num_threads( THREAD_NUM );
32 #pragma omp parallel shared(A,B,C,D) private(i,j)
33  {
34 #pragma omp for schedule(static)
35  for ( i = 0; i < ni; i++ )
36  for ( j = 0; j < nk; j++ )
37  A[i][j] = (float) ( ( i * j + 1 ) % ni ) / ( 5 * ni );
38 
39 #pragma omp for schedule(static)
40  for ( i = 0; i < nk; i++ )
41  for ( j = 0; j < nj; j++ )

43 
44 #pragma omp for schedule(static)
45  for ( i = 0; i < nj; i++ )
46  for ( j = 0; j < nm; j++ )
47  C[i][j] = (float) ( i * ( j + 3 ) % nl ) / ( 5 * nl );
48

51  for ( j = 0; j < nl; j++ )

74 void kernel_3mm( int ni, int nj, int nk, int nl, int nm,
75  float E[ni][nj], float A[ni][nk], float B[nk][nj],
76  float F[nj][nl], float C[nj][nm], float D[nm][nl], float G[ni][nl] )
77 {
78  int i = 0, j = 0, k = 0;
79  omp_set_num_threads( THREAD_NUM );
80 #pragma omp parallel shared(E,A,B,F,C,D,G) private(i,j,k)
81  {
82 #pragma omp for schedule(static)
83  for ( i = 0; i < ni; i++ )
84  for ( j = 0; j < nj; j++ )
85  {
86  E[i][j] = 0.0f;
87  for ( k = 0; k < nk; ++k )
88  E[i][j] += A[i][k] * B[k][j];
89  }
90 #pragma omp for schedule(static)
91  for ( i = 0; i < nj; i++ )
92  for ( j = 0; j < nl; j++ )
93  {
94  F[i][j] = 0.0f;
95  for ( k = 0; k < nm; ++k )
96  F[i][j] += C[i][k] * D[k][j];
97  }
98 
99 #pragma omp for schedule(static)
100  for ( i = 0; i < ni; i++ )
101  for ( j = 0; j < nl; j++ )
102  {
103  G[i][j] = 0.0f;
104  for ( k = 0; k < nj; ++k )
105  G[i][j] += E[i][k] * F[k][j];
106  }
107  }
